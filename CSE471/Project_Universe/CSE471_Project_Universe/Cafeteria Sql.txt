create table public.categories (
  id serial not null,
  name text not null,
  constraint categories_pkey primary key (id)
) TABLESPACE pg_default;

create table public.menu_items (
  name text not null,
  description text null,
  price numeric(8, 2) not null,
  category_id integer null,
  is_special boolean null default false,
  is_available boolean null default true,
  image_url text null,
  tag text null,
  id uuid not null default gen_random_uuid (),
  constraint menu_items_pkey primary key (id),
  constraint menu_items_category_id_fkey foreign KEY (category_id) references categories (id) on delete set null
) TABLESPACE pg_default;


create table public.orders (
  id uuid not null default gen_random_uuid (),
  user_id uuid not null,
  order_date timestamp with time zone null default now(),
  total_amount numeric(10, 2) not null,
  status text not null default 'pending'::text,
  payment_method text null,
  transaction_id text null,
  payment_status text null default 'unpaid'::text,
  delivery_address text null,
  contact_phone text null,
  notes text null,
  created_at timestamp with time zone null default now(),
  scheduled_at timestamp with time zone null default now(),
  constraint orders_pkey primary key (id),
  constraint orders_payment_status_check check (
    (
      payment_status = any (
        array['paid'::text, 'unpaid'::text, 'refunded'::text]
      )
    )
  ),
  constraint orders_status_check check (
    (
      status = any (
        array[
          'pending'::text,
          'completed'::text,
          'cancelled'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;


create table public.order_items (
  id uuid not null default gen_random_uuid (),
  order_id uuid not null,
  item_name text not null default 'Unknown Item'::text,
  quantity integer not null,
  unit_price numeric(10, 2) not null default 0.00,
  price_at_order numeric(10, 2) null,
  menu_item_id uuid null,
  user_id uuid null,
  constraint order_items_pkey primary key (id),
  constraint fk_menu_item foreign KEY (menu_item_id) references menu_items (id) on delete CASCADE,
  constraint order_items_order_id_fkey foreign KEY (order_id) references orders (id) on delete CASCADE,
  constraint order_items_user_id_fkey foreign KEY (user_id) references auth.users (id)
) TABLESPACE pg_default;
create table public.reviews (
  id bigserial not null,
  user_id uuid null,
  menu_item_id uuid null,
  rating integer null,
  comment text null,
  created_at timestamp without time zone null default now(),
  constraint reviews_pkey primary key (id),
  constraint reviews_user_id_fkey foreign KEY (user_id) references auth.users (id),
  constraint reviews_rating_check check (
    (
      (rating >= 1)
      and (rating <= 5)
    )
  )
) TABLESPACE pg_default;


create table public.suggestions (
  id bigserial not null,
  user_id uuid null,
  title text not null,
  description text null,
  created_at timestamp without time zone null default now(),
  constraint suggestions_pkey primary key (id),
  constraint suggestions_user_id_fkey foreign KEY (user_id) references auth.users (id)
) TABLESPACE pg_default;

create table public.cart_items (
  id bigint generated by default as identity not null,
  user_id uuid not null,
  menu_item_id integer not null,
  quantity integer not null default 1,
  inserted_at timestamp with time zone null default now(),
  constraint cart_items_pkey primary key (id),
  constraint cart_items_user_id_fkey foreign KEY (user_id) references auth.users (id)
) TABLESPACE pg_default;


create table public.favorites (
  user_id uuid not null,
  menu_item_id uuid not null,
  constraint favorites_pkey primary key (user_id, menu_item_id),
  constraint favorites_user_id_fkey foreign KEY (user_id) references auth.users (id)
) TABLESPACE pg_default;

create table public.feedback (
  id bigserial not null,
  user_id uuid null,
  content text not null,
  created_at timestamp without time zone null default now(),
  constraint feedback_pkey primary key (id),
  constraint feedback_user_id_fkey foreign KEY (user_id) references auth.users (id)
) TABLESPACE pg_default;
