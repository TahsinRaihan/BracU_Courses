-- Create RLS policies for orders table
CREATE POLICY "anon_insert_orders" ON public.orders
FOR INSERT TO anon, authenticated
WITH CHECK (true); -- Allows any user (including anon) to insert

CREATE POLICY "anon_select_orders" ON public.orders
FOR SELECT TO authenticated
USING (user_id = (SELECT auth.uid())); -- Only allow users to see their own orders (if authenticated)

-- Create RLS policies for order_items table
CREATE POLICY "anon_insert_order_items" ON public.order_items
FOR INSERT TO anon, authenticated
WITH CHECK (true); -- Allows any user (including anon) to insert

CREATE POLICY "anon_select_order_items" ON public.order_items
FOR SELECT TO authenticated
USING (EXISTS (
    SELECT 1 
    FROM public.orders 
    WHERE id = order_id 
    AND user_id = (SELECT auth.uid())
)); -- Only allow users to see order items for their own orders

-- Grant necessary permissions
GRANT ALL ON public.orders TO authenticated;
GRANT ALL ON public.order_items TO authenticated;


DROP POLICY IF EXISTS "anon_insert_orders" ON public.orders;
DROP POLICY IF EXISTS "anon_insert_order_items" ON public.order_items;

-- Create new RLS policy for orders (INSERT)
CREATE POLICY "anon_and_authenticated_insert_orders" ON public.orders
FOR INSERT WITH CHECK (
  (auth.uid() IS NOT NULL AND user_id = auth.uid()) OR -- Authenticated user inserting their own order
  (auth.role() = 'anon' AND user_id IS NOT NULL) -- Anonymous user inserting an order (user_id will be the anonymous UUID)
);

-- Create new RLS policy for order_items (INSERT)
-- This policy assumes that order_id is already linked to a user_id in the orders table
CREATE POLICY "anon_and_authenticated_insert_order_items" ON public.order_items
FOR INSERT WITH CHECK (
  EXISTS (
    SELECT 1 FROM public.orders
    WHERE orders.id = order_id
    AND (
      (auth.uid() IS NOT NULL AND orders.user_id = auth.uid()) OR -- Authenticated user's order
      (auth.role() = 'anon' AND orders.user_id IS NOT NULL) -- Anonymous user's order
    )
  )
);
